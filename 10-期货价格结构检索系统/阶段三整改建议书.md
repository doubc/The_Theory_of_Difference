```python
# src/models.py (修订版)
"""
价格结构形式系统 — 对象模型定义
Point → Segment → Zone → Cycle → Structure → Bundle

设计原则：
1. 对象只保存状态，不做复杂计算（计算放在 relations / invariants）
2. 所有对象支持 to_dict / from_dict（JSON 友好）
3. Point 只保留 (t, x, idx)，其它属性由算子推导
"""
from __future__ import annotations
import math
from dataclasses import dataclass, field, asdict
from enum import Enum
from typing import Optional, Any
from datetime import datetime


# ─── 枚举 ──────────────────────────────────────────────────
class Direction(Enum):
    UP = 1
    DOWN = -1
    FLAT = 0

    @classmethod
    def from_delta(cls, delta: float, eps: float = 1e-9) -> "Direction":
        if delta > eps:
            return cls.UP
        if delta < -eps:
            return cls.DOWN
        return cls.FLAT


class ZoneSource(Enum):
    HIGH_CLUSTER = "high_cluster"
    LOW_CLUSTER = "low_cluster"
    PIVOT = "pivot"
    VOLUME = "volume"


class Phase(Enum):
    FORMATION = "formation"
    CONFIRMATION = "confirmation"
    BREAKDOWN = "breakdown"
    INVERSION = "inversion"


# ─── Point ─────────────────────────────────────────────────
@dataclass
class Point:
    t: datetime
    x: float
    idx: int = 0  # 在原始 Bar 序列中的索引

    @property
    def log_x(self) -> float:
        return math.log(self.x) if self.x > 0 else float("-inf")

    def to_dict(self) -> dict:
        return {"t": self.t.isoformat(), "x": self.x, "idx": self.idx}

    @classmethod
    def from_dict(cls, d: dict) -> "Point":
        return cls(t=datetime.fromisoformat(d["t"]), x=d["x"], idx=d.get("idx", 0))

    def __repr__(self):
        return f"Point(t={self.t:%Y-%m-%d}, x={self.x:.2f})"


# ─── Segment ───────────────────────────────────────────────
@dataclass
class Segment:
    start: Point
    end: Point
    duration: float = 0.0
    delta: float = 0.0
    log_delta: float = 0.0
    avg_rate: float = 0.0
    direction: Direction = Direction.FLAT
    noise_level: float = 0.0

    def __post_init__(self):
        # 用时间戳差计算，避免 duration 初值 0 误判
        if self.start.t != self.end.t:
            self.duration = (self.end.t - self.start.t).days or
            (self.end.t - self.start.t).total_seconds() / 86400

    self.delta = self.end.x - self.start.x
    self.log_delta = self.end.log_x - self.start.log_x
    if self.duration > 0:
        self.avg_rate = self.delta / self.duration
    self.direction = Direction.from_delta(self.delta)


@property
def abs_delta(self) -> float:
    return abs(self.delta)


@property
def abs_rate(self) -> float:
    return abs(self.avg_rate)


@property
def log_rate(self) -> float:
    """对数速率 — 跨品种可比"""
    return self.log_delta / self.duration if self.duration > 0 else 0.0


def to_dict(self) -> dict:
    return {
        "start": self.start.to_dict(),
        "end": self.end.to_dict(),
        "duration": self.duration,
        "delta": self.delta,
        "log_delta": self.log_delta,
        "avg_rate": self.avg_rate,
        "direction": self.direction.value,
        "noise_level": self.noise_level,
    }


@classmethod
def from_dict(cls, d: dict) -> "Segment":
    return cls(
        start=Point.from_dict(d["start"]),
        end=Point.from_dict(d["end"]),
    )


def __repr__(self):
    arrow = {Direction.UP: "↑", Direction.DOWN: "↓", Direction.FLAT: "→"}[self.direction]
    return f"Segment({arrow} {self.delta:+.2f} in {self.duration:.0f}d)"


# ─── Zone ──────────────────────────────────────────────────
@dataclass
class Zone:
    price_center: float
    bandwidth: float  # 半宽度
    source: ZoneSource = ZoneSource.PIVOT
    strength: float = 0.0
    touches: list[Point] = field(default_factory=list)

    @property
    def upper(self) -> float:
        return self.price_center + self.bandwidth

    @property
    def lower(self) -> float:
        return self.price_center - self.bandwidth

    @property
    def relative_bandwidth(self) -> float:
        """相对带宽 (跨品种可比)"""
        return self.bandwidth / self.price_center if self.price_center > 0 else 0.0

    def contains(self, price: float) -> bool:
        return self.lower <= price <= self.upper

    def distance_to(self, price: float) -> float:
        if self.contains(price):
            return 0.0
        return price - self.upper if price > self.upper else self.lower - price

    def compute_strength(self, decay: float = 0.9) -> float:
        """strength = Σ decay^i  (试探次数 × 距离衰减)"""
        self.strength = sum(decay ** i for i in range(len(self.touches)))
        return self.strength

    def to_dict(self) -> dict:
        return {
            "price_center": self.price_center,
            "bandwidth": self.bandwidth,
            "source": self.source.value,
            "strength": self.strength,
            "touches": [p.to_dict() for p in self.touches],
        }

    def __repr__(self):
        return f"Zone(center={self.price_center:.2f}, bw=±{self.bandwidth:.2f}, strength={self.strength:.1f})"


# ─── Cycle ─────────────────────────────────────────────────
@dataclass
class Cycle:
    entry: Segment
    exit: Segment
    zone: Zone

    @property
    def speed_ratio(self) -> float:
        if self.entry.abs_rate == 0 or self.exit.abs_rate == 0:
            return 0.0
        return self.exit.abs_rate / self.entry.abs_rate

    @property
    def log_speed_ratio(self) -> float:
        """对数速度比 — 品种无关"""
        le, lx = abs(self.entry.log_rate), abs(self.exit.log_rate)
        if le == 0 or lx == 0:
            return 0.0
        return lx / le

    @property
    def time_ratio(self) -> float:
        if self.entry.duration == 0 or self.exit.duration == 0:
            return 0.0
        return self.entry.duration / self.exit.duration

    @property
    def amplitude_ratio(self) -> float:
        if self.entry.abs_delta == 0:
            return 0.0
        return self.exit.abs_delta / self.entry.abs_delta

    def to_dict(self) -> dict:
        return {
            "entry": self.entry.to_dict(),
            "exit": self.exit.to_dict(),
            "zone": self.zone.to_dict(),
        }

    def __repr__(self):
        return (f"Cycle(speed_r={self.speed_ratio:.2f}, "
                f"time_r={self.time_ratio:.2f}, amp_r={self.amplitude_ratio:.2f})")


# ─── Structure ─────────────────────────────────────────────
@dataclass
class Structure:
    zone: Zone
    cycles: list[Cycle] = field(default_factory=list)
    phases: list[Phase] = field(default_factory=list)
    invariants: dict = field(default_factory=dict)
    typicality: float = 0.0
    label: Optional[str] = None  # 规则引擎打的标签
    symbol: Optional[str] = None  # 品种
    t_start: Optional[datetime] = None
    t_end: Optional[datetime] = None

    @property
    def cycle_count(self) -> int:
        return len(self.cycles)

    def signature(self) -> str:
        """结构签名 — 用于检索与去重"""
        parts = [
            f"n={self.cycle_count}",
            f"sr={self.avg_speed_ratio:.2f}",
            f"tr={self.avg_time_ratio:.2f}",
            f"zbw={self.zone.relative_bandwidth:.3f}",
        ]
        return "|".join(parts)

    @property
    def avg_speed_ratio(self) -> float:
        if not self.cycles:
            return 0.0
        return sum(c.speed_ratio for c in self.cycles) / len(self.cycles)

    @property
    def avg_time_ratio(self) -> float:
        if not self.cycles:
            return 0.0
        return sum(c.time_ratio for c in self.cycles) / len(self.cycles)

    def to_dict(self) -> dict:
        return {
            "zone": self.zone.to_dict(),
            "cycles": [c.to_dict() for c in self.cycles],
            "phases": [p.value for p in self.phases],
            "invariants": self.invariants,
            "typicality": self.typicality,
            "label": self.label,
            "symbol": self.symbol,
            "t_start": self.t_start.isoformat() if self.t_start else None,
            "t_end": self.t_end.isoformat() if self.t_end else None,
            "signature": self.signature(),
        }

    def __repr__(self):
        return (f"Structure(label={self.label}, zone={self.zone}, "
                f"cycles={self.cycle_count}, typ={self.typicality:.2f})")


@dataclass
class Bundle:
    structures: list[Structure] = field(default_factory=list)
    generator_constraint: str = ""

    def __repr__(self):
        return f"Bundle('{self.generator_constraint}', n={len(self.structures)})"
```

```python
# src/relations.py
"""
关系算子层 — 不存储，纯计算
所有"点间/段间/结构间"的关系在这里计算
"""
from __future__ import annotations
import math
from typing import Sequence
from .models import Point, Segment, Zone, Cycle, Structure


# ─── 点间算子 ──────────────────────────────────────────────
def first_diff(p1: Point, p2: Point) -> float:
    return p2.x - p1.x


def log_diff(p1: Point, p2: Point) -> float:
    """对数差分 — 品种无关"""
    return p2.log_x - p1.log_x


def second_diff(p1: Point, p2: Point, p3: Point) -> float:
    return p3.x - 2 * p2.x + p1.x


def time_gap(p1: Point, p2: Point) -> float:
    return (p2.t - p1.t).days


# ─── 段间算子 ──────────────────────────────────────────────
def speed_ratio(s1: Segment, s2: Segment) -> float:
    if s1.abs_rate == 0:
        return float("inf") if s2.abs_rate > 0 else 0.0
    return s2.abs_rate / s1.abs_rate


def time_ratio(s1: Segment, s2: Segment) -> float:
    if s2.duration == 0:
        return 0.0
    return s1.duration / s2.duration


def direction_change(s1: Segment, s2: Segment) -> int:
    """方向变化：0=同向, 1=反向, -1=含FLAT"""
    if s1.direction.value * s2.direction.value < 0:
        return 1
    if s1.direction.value * s2.direction.value > 0:
        return 0
    return -1


# ─── 点与区 ────────────────────────────────────────────────
def distance_to_zone(p: Point, z: Zone) -> float:
    return z.distance_to(p.x)


def relative_distance_to_zone(p: Point, z: Zone) -> float:
    """相对距离 — 以 Zone 带宽为单位"""
    if z.bandwidth == 0:
        return 0.0
    return z.distance_to(p.x) / z.bandwidth


# ─── 极值点聚集度 ──────────────────────────────────────────
def extrema_dispersion(points: Sequence[Point]) -> float:
    """极值点价格的相对标准差 (CV)"""
    if len(points) < 2:
        return 0.0
    prices = [p.x for p in points]
    mean = sum(prices) / len(prices)
    if mean == 0:
        return 0.0
    var = sum((x - mean) ** 2 for x in prices) / len(prices)
    return math.sqrt(var) / mean


def extrema_trend(points: Sequence[Point]) -> float:
    """极值点趋势斜率（线性回归斜率 / 均值）"""
    if len(points) < 2:
        return 0.0
    n = len(points)
    xs = list(range(n))
    ys = [p.x for p in points]
    mx = sum(xs) / n
    my = sum(ys) / n
    num = sum((xs[i] - mx) * (ys[i] - my) for i in range(n))
    den = sum((xs[i] - mx) ** 2 for i in range(n))
    if den == 0 or my == 0:
        return 0.0
    return (num / den) / my


# ─── 结构级不变量 ──────────────────────────────────────────
def structure_invariants(s: Structure) -> dict:
    """提取结构的全部关系不变量 — 供检索与分类使用"""
    highs, lows = [], []
    for c in s.cycles:
        for p in (c.entry.start, c.entry.end, c.exit.start, c.exit.end):
            if p.x >= s.zone.price_center:
                highs.append(p)
            else:
                lows.append(p)
    return {
        "cycle_count": s.cycle_count,
        "avg_speed_ratio": s.avg_speed_ratio,
        "avg_time_ratio": s.avg_time_ratio,
        "high_dispersion": extrema_dispersion(highs),
        "low_dispersion": extrema_dispersion(lows),
        "high_trend": extrema_trend(highs),
        "low_trend": extrema_trend(lows),
        "zone_rel_bw": s.zone.relative_bandwidth,
        "zone_strength": s.zone.strength,
    }
```

### **部门三：编译器 — 极值提取（新文件 compiler/pivots.py）**

用 **ZigZag** 算法（基于最小摆动幅度），这是最稳的起点。

```python
# src/compiler/pivots.py
"""
极值点提取 — ZigZag 算法（基于最小摆动幅度）
可选：ATR-based 自适应阈值
"""
from __future__ import annotations
from typing import Sequence
from datetime import datetime
from ..models import Point
from ..data.loader import Bar


def extract_pivots_zigzag(
        bars: Sequence[Bar],
        min_amplitude_pct: float = 0.03,  # 最小摆动幅度 3%
        min_duration_days: int = 3,  # 最小持续天数
        use_log: bool = True,  # 对数空间判断
) -> list[Point]:
    """
    ZigZag 极值提取
    
    算法：
    1. 从第一个 bar 开始，维护 "候选极值"
    2. 价格在候选点相反方向运动 > min_amplitude_pct 时确认候选
    3. 否则继续更新候选极值
    
    返回：Point 列表（交替的高/低点）
    """
    if len(bars) < 2:
        return []

    import math
    def price_diff(a: float, b: float) -> float:
        if use_log and a > 0 and b > 0:
            return math.log(b) - math.log(a)
        return (b - a) / a if a != 0 else 0.0

    pivots: list[Point] = []
    # 初始化
    anchor_idx = 0
    anchor_price = bars[0].close
    direction = 0  # 0=未定, 1=上升, -1=下降
    candidate_idx = 0
    candidate_price = bars[0].close

    for i in range(1, len(bars)):
        price_h = bars[i].high
        price_l = bars[i].low

        if direction >= 0:
            # 记录新高
            if price_h > candidate_price:
                candidate_idx = i
                candidate_price = price_h
            # 从 candidate 下跌幅度足够，确认候选为 pivot
            move_down = price_diff(candidate_price, price_l)
            dur = (bars[i].timestamp - bars[candidate_idx].timestamp).days
            if move_down <= -min_amplitude_pct and dur >= min_duration_days:
                # 确认高点
                pivots.append(Point(t=bars[candidate_idx].timestamp,
                                    x=candidate_price,
                                    idx=candidate_idx))
                direction = -1
                anchor_idx = candidate_idx
                anchor_price = candidate_price
                candidate_idx = i
                candidate_price = price_l

        if direction <= 0:
            if price_l < candidate_price or direction == 0:
                if direction == -1 or (direction == 0 and price_l < candidate_price):
                    candidate_idx = i
                    candidate_price = price_l
            move_up = price_diff(candidate_price, price_h)
            dur = (bars[i].timestamp - bars[candidate_idx].timestamp).days
            if move_up >= min_amplitude_pct and dur >= min_duration_days:
                pivots.append(Point(t=bars[candidate_idx].timestamp,
                                    x=candidate_price,
                                    idx=candidate_idx))
                direction = 1
                anchor_idx = candidate_idx
                anchor_price = candidate_price
                candidate_idx = i
                candidate_price = price_h

    # 收尾：把最后的候选点加入
    if candidate_idx != anchor_idx:
        pivots.append(Point(t=bars[candidate_idx].timestamp,
                            x=candidate_price,
                            idx=candidate_idx))

    return pivots


def extract_pivots_atr(
        bars: Sequence[Bar],
        atr_window: int = 14,
        atr_multiplier: float = 2.0,
        min_duration_days: int = 3,
) -> list[Point]:
    """ATR 自适应阈值 — 波动大的时期阈值大，小的时期阈值小"""
    if len(bars) < atr_window + 2:
        return []
    # 计算 ATR
    trs = []
    for i in range(1, len(bars)):
        tr = max(
            bars[i].high - bars[i].low,
            abs(bars[i].high - bars[i - 1].close),
            abs(bars[i].low - bars[i - 1].close),
        )
        trs.append(tr)
    # 对每个位置计算自适应阈值，然后类似 zigzag
    # 简化：用全局 ATR 均值作为阈值
    atr = sum(trs) / len(trs)
    # 退化为按绝对幅度的 zigzag
    threshold = atr * atr_multiplier
    avg_price = sum(b.close for b in bars) / len(bars)
    return extract_pivots_zigzag(
        bars,
        min_amplitude_pct=threshold / avg_price,
        min_duration_days=min_duration_days,
        use_log=False,
    )
```

```python
# src/compiler/segments.py
"""
段生成 — 把相邻极值点连接成 Segment
"""
from __future__ import annotations
from typing import Sequence
from ..models import Point, Segment


def build_segments(pivots: Sequence[Point]) -> list[Segment]:
    """相邻 pivot 两两成段"""
    if len(pivots) < 2:
        return []
    return [Segment(start=pivots[i], end=pivots[i + 1])
            for i in range(len(pivots) - 1)]


def merge_micro_segments(
        segments: list[Segment],
        min_abs_delta_pct: float = 0.005,
) -> list[Segment]:
    """
    合并微小段 — 相对幅度 < 阈值的段，与相邻段合并
    防止高频噪声污染结构识别
    """
    if not segments:
        return []
    result = [segments[0]]
    for seg in segments[1:]:
        last = result[-1]
        rel_delta = seg.abs_delta / seg.start.x if seg.start.x > 0 else 0
        if rel_delta < min_abs_delta_pct:
            # 跳过 seg，扩展 last 的 end
            result[-1] = Segment(start=last.start, end=seg.end)
        else:
            result.append(seg)
    return result
```

```python
"""
关键区识别 — 对极值点做价格聚类
方法：1D 密度聚类（Mean Shift 简化版）
"""
from __future__ import annotations
import math
from typing import Sequence
from ..models import Point, Zone, ZoneSource


def cluster_prices_1d(
        prices: Sequence[float],
        bandwidth_pct: float = 0.02,
        min_cluster_size: int = 2,
) -> list[tuple[float, list[int]]]:
    """
    简易 1D 均值漂移聚类
    
    返回：[(cluster_center, [原始索引]), ...]
    """
    if len(prices) < min_cluster_size:
        return []

    clusters: list[tuple[float, list[int]]] = []
    sorted_idx = sorted(range(len(prices)), key=lambda i: prices[i])

    for i in sorted_idx:
        p = prices[i]
        bw = p * bandwidth_pct
        placed = False
        for ci, (center, members) in enumerate(clusters):
            if abs(p - center) <= bw:
                # 加入并更新中心
                members.append(i)
                new_center = sum(prices[m] for m in members) / len(members)
                clusters[ci] = (new_center, members)
                placed = True
                break
        if not placed:
            clusters.append((p, [i]))

    # 过滤最小簇
    return [(c, m) for c, m in clusters if len(m) >= min_cluster_size]


def detect_zones_from_pivots(
        pivots: Sequence[Point],
        bandwidth_pct: float = 0.02,
        min_touches: int = 2,
        separate_high_low: bool = True,
) -> list[Zone]:
    """
    从极值点中识别 Zone
    
    - separate_high_low: 分开识别高点聚集区(压力)和低点聚集区(支撑)
                         判定依据：局部位置是高还是低
    """
    if len(pivots) < min_touches:
        return []

    if separate_high_low:
        # 判定每个 pivot 是局部高还是局部低：看前后邻居
        highs, lows = [], []
        for i, p in enumerate(pivots):
            prev_x = pivots[i - 1].x if i > 0 else p.x
            next_x = pivots[i + 1].x if i < len(pivots) - 1 else p.x
            if p.x > prev_x and p.x > next_x:
                highs.append((i, p))
            elif p.x < prev_x and p.x < next_x:
                lows.append((i, p))
            else:
                # 端点或平点：按相对位置判
                if p.x >= (prev_x + next_x) / 2:
                    highs.append((i, p))
                else:
                    lows.append((i, p))

        zones: list[Zone] = []
        for group, source in [(highs, ZoneSource.HIGH_CLUSTER),
                              (lows, ZoneSource.LOW_CLUSTER)]:
            if len(group) < min_touches:
                continue
            prices = [p.x for _, p in group]
            clusters = cluster_prices_1d(prices, bandwidth_pct, min_touches)
            for center, member_idx in clusters:
                touches = [group[mi][1] for mi in member_idx]
                bw = center * bandwidth_pct
                z = Zone(
                    price_center=center,
                    bandwidth=bw,
                    source=source,
                    touches=touches,
                )
                z.compute_strength()
                zones.append(z)
        return zones
    else:
        prices = [p.x for p in pivots]
        clusters = cluster_prices_1d(prices, bandwidth_pct, min_touches)
        zones = []
        for center, member_idx in clusters:
            touches = [pivots[mi] for mi in member_idx]
            z = Zone(
                price_center=center,
                bandwidth=center * bandwidth_pct,
                source=ZoneSource.PIVOT,
                touches=touches,
            )
            z.compute_strength()
            zones.append(z)
        return zones
```

```python
# src/compiler/cycles.py
"""
Cycle 与 Structure 组装
"""
from __future__ import annotations
from typing import Sequence
from ..models import Point, Segment, Zone, Cycle, Structure
from ..relations import structure_invariants


def build_cycles_around_zone(
        segments: Sequence[Segment],
        zone: Zone,
) -> list[Cycle]:
    """
    识别围绕 zone 的 Cycle
    
    定义：一个 Cycle = 进入 zone 的段 + 离开 zone 的段
    具体：连续两个段，其中 entry.end 触及 zone，exit.start 从 zone 离开
    """
    cycles: list[Cycle] = []
    for i in range(len(segments) - 1):
        entry = segments[i]
        exit_ = segments[i + 1]
        # entry 终点需落在 zone 或接近 zone
        if not _touches_zone(entry.end, zone):
            continue
        # 方向需反向（进入后离开）
        if entry.direction.value * exit_.direction.value >= 0:
            continue
        cycles.append(Cycle(entry=entry, exit=exit_, zone=zone))
    return cycles


def _touches_zone(p: Point, zone: Zone, slack: float = 0.005) -> bool:
    """点是否触及 zone（允许小幅越界）"""
    slack_abs = zone.price_center * slack
    return (zone.lower - slack_abs) <= p.x <= (zone.upper + slack_abs)


def assemble_structures(
        segments: Sequence[Segment],
        zones: Sequence[Zone],
        min_cycles: int = 2,
        symbol: str | None = None,
) -> list[Structure]:
    """对每个 zone 组装 Structure 候选"""
    structures: list[Structure] = []
    for zone in zones:
        cycles = build_cycles_around_zone(segments, zone)
        if len(cycles) < min_cycles:
            continue
        s = Structure(
            zone=zone,
            cycles=cycles,
            symbol=symbol,
            t_start=cycles[0].entry.start.t,
            t_end=cycles[-1].exit.end.t,
        )
        s.invariants = structure_invariants(s)
        structures.append(s)
    return structures
```

```python
# src/compiler/pipeline.py
"""
结构编译器 — 统一入口
价格序列 → 结构对象
"""
from __future__ import annotations
from dataclasses import dataclass
from typing import Sequence
from ..data.loader import Bar
from ..models import Structure, Zone, Segment, Point
from .pivots import extract_pivots_zigzag, extract_pivots_atr
from .segments import build_segments, merge_micro_segments
from .zones import detect_zones_from_pivots
from .cycles import assemble_structures


@dataclass
class CompileConfig:
    # 极值提取
    pivot_method: str = "zigzag"  # zigzag / atr
    min_amplitude_pct: float = 0.03
    min_duration_days: int = 3
    use_log_price: bool = True
    atr_window: int = 14
    atr_multiplier: float = 2.0

    # 段过滤
    min_segment_delta_pct: float = 0.005

    # Zone
    zone_bandwidth_pct: float = 0.02
    min_zone_touches: int = 2
    separate_high_low: bool = True

    # Structure
    min_cycles: int = 2


@dataclass
class CompileResult:
    pivots: list[Point]
    segments: list[Segment]
    zones: list[Zone]
    structures: list[Structure]


def compile_structures(
        bars: Sequence[Bar],
        config: CompileConfig | None = None,
        symbol: str | None = None,
) -> CompileResult:
    cfg = config or CompileConfig()

    # Step 1: 极值
    if cfg.pivot_method == "zigzag":
        pivots = extract_pivots_zigzag(
            bars,
            min_amplitude_pct=cfg.min_amplitude_pct,
            min_duration_days=cfg.min_duration_days,
            use_log=cfg.use_log_price,
        )
    else:
        pivots = extract_pivots_atr(
            bars,
            atr_window=cfg.atr_window,
            atr_multiplier=cfg.atr_multiplier,
            min_duration_days=cfg.min_duration_days,
        )

    # Step 2: 段
    segments = build_segments(pivots)
    segments = merge_micro_segments(segments, cfg.min_segment_delta_pct)

    # Step 3: Zone
    zones = detect_zones_from_pivots(
        pivots,
        bandwidth_pct=cfg.zone_bandwidth_pct,
        min_touches=cfg.min_zone_touches,
        separate_high_low=cfg.separate_high_low,
    )

    # Step 4: Structure
    sym = symbol or (bars[0].symbol if bars else None)
    structures = assemble_structures(
        segments, zones,
        min_cycles=cfg.min_cycles,
        symbol=sym,
    )

    return CompileResult(
        pivots=pivots,
        segments=segments,
        zones=zones,
        structures=structures,
    )
```

```python
# src/dsl/rule.py
"""
规则引擎 — 基于 YAML 的结构规则定义与匹配
"""
from __future__ import annotations
import yaml
from dataclasses import dataclass, field
from typing import Any, Callable
from ..models import Structure


# ─── 约束原语 ──────────────────────────────────────────────
def _cmp(value: float, spec: Any) -> bool:
    """spec 可以是：
       单值 -> value == spec
       [lo, hi] -> lo <= value <= hi
       {"gt": x} / {"lt": x} / {"gte": x} / {"lte": x}
       {"between": [lo, hi]}
    """
    if isinstance(spec, (int, float)):
        return abs(value - spec) < 1e-9
    if isinstance(spec, list) and len(spec) == 2:
        return spec[0] <= value <= spec[1]
    if isinstance(spec, dict):
        if "gt" in spec and not value > spec["gt"]: return False
        if "gte" in spec and not value >= spec["gte"]: return False
        if "lt" in spec and not value < spec["lt"]: return False
        if "lte" in spec and not value <= spec["lte"]: return False
        if "between" in spec:
            lo, hi = spec["between"]
            if not (lo <= value <= hi): return False
        return True
    return False


# ─── 规则定义 ──────────────────────────────────────────────
@dataclass
class Rule:
    name: str
    zone_source: str | None = None  # high_cluster / low_cluster
    cycles: Any = None  # e.g. {"gte": 3}
    speed_ratio: Any = None  # e.g. {"gt": 1.5}
    time_ratio: Any = None  # e.g. {"gt": 1.8}
    high_dispersion: Any = None  # e.g. {"lt": 0.02}
    low_dispersion: Any = None
    high_trend: Any = None
    low_trend: Any = None
    zone_rel_bw: Any = None
    zone_strength: Any = None
    description: str = ""

    def match(self, s: Structure) -> tuple[bool, dict]:
        """
        匹配结构
        返回 (是否匹配, 每个约束的通过情况)
        """
        checks = {}

        if self.zone_source is not None:
            checks["zone_source"] = s.zone.source.value == self.zone_source

        inv = s.invariants or {}
        attr_map = {
            "cycles": inv.get("cycle_count", s.cycle_count),
            "speed_ratio": inv.get("avg_speed_ratio", s.avg_speed_ratio),
            "time_ratio": inv.get("avg_time_ratio", s.avg_time_ratio),
            "high_dispersion": inv.get("high_dispersion", 0),
            "low_dispersion": inv.get("low_dispersion", 0),
            "high_trend": inv.get("high_trend", 0),
            "low_trend": inv.get("low_trend", 0),
            "zone_rel_bw": inv.get("zone_rel_bw", s.zone.relative_bandwidth),
            "zone_strength": inv.get("zone_strength", s.zone.strength),
        }
        for key, value in attr_map.items():
            spec = getattr(self, key, None)
            if spec is not None:
                checks[key] = _cmp(value, spec)

        passed = all(checks.values())
        return passed, checks

    def typicality_score(self, checks: dict) -> float:
        """简单的典型度：通过的比例"""
        if not checks:
            return 0.0
        return sum(1 for v in checks.values() if v) / len(checks)


# ─── 规则加载器 ────────────────────────────────────────────
def load_rules(yaml_path: str) -> list[Rule]:
    with open(yaml_path) as f:
        data = yaml.safe_load(f)
    rules = []
    for item in data.get("rules", []):
        rules.append(Rule(**item))
    return rules


# ─── 扫描器 ────────────────────────────────────────────────
@dataclass
class RuleMatch:
    rule: Rule
    structure: Structure
    checks: dict
    typicality: float


def scan(structures: list[Structure], rules: list[Rule]) -> list[RuleMatch]:
    matches: list[RuleMatch] = []
    for s in structures:
        for r in rules:
            passed, checks = r.match(s)
            if passed:
                typ = r.typicality_score(checks)
                s.label = r.name
                s.typicality = typ
                matches.append(RuleMatch(rule=r, structure=s, checks=checks, typicality=typ))
    return matches
```

```python
from __future__ import annotations
from pathlib import Path
from datetime import datetime
from typing import Sequence
from .loader import Bar, bar_to_dict


class ParquetCache:
    def __init__(self, cache_dir: str = "data/cache"):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)

    def path(self, symbol: str, freq: str = "1d") -> Path:
        return self.cache_dir / f"{symbol}_{freq}.parquet"

    def save(self, symbol: str, bars: Sequence[Bar], freq: str = "1d"):
        try:
            import pandas as pd
        except ImportError:
            raise RuntimeError("需要 pandas + pyarrow")
        df = pd.DataFrame([bar_to_dict(b) for b in bars])
        df.to_parquet(self.path(symbol, freq), index=False)

    def load(self, symbol: str, freq: str = "1d") -> list[Bar] | None:
        p = self.path(symbol, freq)
        if not p.exists():
            return None
        try:
            import pandas as pd
        except ImportError:
            return None
        df = pd.read_parquet(p)
        return [
            Bar(
                symbol=row["symbol"],
                timestamp=row["timestamp"].to_pydatetime() if hasattr(row["timestamp"], "to_pydatetime") else row[
                    "timestamp"],
                open=row["open"], high=row["high"], low=row["low"],
                close=row["close"], volume=row["volume"],
                open_interest=row.get("open_interest"),
                adj_factor=row.get("adj_factor"),
            )
            for _, row in df.iterrows()
        ]
```

```python
            typicality=d.get("typicality", 0.0),
            annotation=d.get("annotation", ""),
            forward_outcome=d.get("forward_outcome"),
        )


class SampleStore:
    def __init__(self, path: str = "data/samples/library.jsonl"):
        self.path = Path(path)
        self.path.parent.mkdir(parents=True, exist_ok=True)

    def append(self, sample: Sample):
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(sample.to_json() + "\n")

    def load_all(self) -> list[Sample]:
        if not self.path.exists():
            return []
        samples = []
        with open(self.path, encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line:
                    samples.append(Sample.from_json(line))
        return samples

    def filter(self, label_type: str | None = None, symbol: str | None = None) -> list[Sample]:
        all_ = self.load_all()
        if label_type:
            all_ = [s for s in all_ if s.label_type == label_type]
        if symbol:
            all_ = [s for s in all_ if s.symbol == symbol]
        return all_

    def sample_from_structure(
            s: Structure,
            label_type: str,
            sample_id: str,
            annotation: str = "",
            forward: ForwardOutcome | None = None,

            t_end=s.t_end or datetime.now(),
            structure=s.to_dict(),
            label_type=label_type,
            label_phase=s.phases[-1].value if s.phases else "formation",
            typicality=s.typicality,
            annotation=annotation,
            forward_outcome=asdict(forward) if forward else None,
        )
```

```python
# src/sample/outcome.py
"""
给样本打 forward_outcome 标签
— 基于结构结束后的 N 日价格表现
"""
from __future__ import annotations
from datetime import datetime, timedelta
from typing import Sequence
from ..data.loader import Bar
from .store import ForwardOutcome


def compute_forward_outcome(
        bars: Sequence[Bar],
        t_end: datetime,
        windows: tuple[int, ...] = (5, 10, 20),
) -> ForwardOutcome:
    """计算结构结束后的后续表现"""
    # 找到 t_end 对应的索引
    idx = next((i for i, b in enumerate(bars) if b.timestamp >= t_end), None)
    if idx is None or idx >= len(bars) - 1:
        return ForwardOutcome()

    base = bars[idx].close
    outcome = ForwardOutcome()

    for w in windows:
        end_idx = min(idx + w, len(bars) - 1)
        if end_idx > idx:
            ret = (bars[end_idx].close - base) / base
            setattr(outcome, f"ret_{w}d", ret)

    # 20日内最大涨幅 / 最大回撤
    end_idx = min(idx + 20, len(bars) - 1)
    window_bars = bars[idx + 1: end_idx + 1]
    if window_bars:
        max_p = max(b.high for b in window_bars)
        min_p = min(b.low for b in window_bars)
        outcome.max_rise_20d = (max_p - base) / base
        outcome.max_dd_20d = (min_p - base) / base

    return outcome
```

### **部门十一：相似性检索（retrieval/similarity.py）**

```python
# src/retrieval/similarity.py
"""
结构相似性度量 — 三层
1. 几何相似（基于不变量向量）
2. 关系相似（基于离散约束一致性）
3. 结构族相似（是否同一 Bundle / 镜像）
"""
from __future__ import annotations
import math
from dataclasses import dataclass
from ..models import Structure


@dataclass
class SimilarityScore:
    total: float
    geometric: float
    relational: float
    family: float
    matched_invariants: dict


# ─── 几何相似 — 不变量向量的余弦/欧氏距离 ──────────────────
INVARIANT_KEYS = [
    "cycle_count",
    "avg_speed_ratio",
    "avg_time_ratio",
    "high_dispersion",
    "low_dispersion",
    "zone_rel_bw",
    "zone_strength",
]

# 每个不变量的典型量级，用于标准化
INVARIANT_SCALES = {
    "cycle_count": 5.0,
    "avg_speed_ratio": 2.0,
    "avg_time_ratio": 2.0,
    "high_dispersion": 0.05,
    "low_dispersion": 0.05,
    "zone_rel_bw": 0.05,
    "zone_strength": 5.0,
}


def _normalized_vector(inv: dict) -> list[float]:
    return [
        (inv.get(k, 0) or 0) / INVARIANT_SCALES[k]
        for k in INVARIANT_KEYS
    ]


def geometric_similarity(s1: Structure, s2: Structure) -> float:
    """归一化向量的 1 - 欧氏距离(clip 到 [0,1])"""
    v1 = _normalized_vector(s1.invariants)
    v2 = _normalized_vector(s2.invariants)
    d = math.sqrt(sum((a - b) ** 2 for a, b in zip(v1, v2)))
    return max(0.0, 1.0 - d / math.sqrt(len(v1)))


# ─── 关系相似 — 方向序列 / Zone 来源一致 ────────────────────
def relational_similarity(s1: Structure, s2: Structure) -> float:
    score = 0.0
    n = 0

    # Zone 来源一致性
    n += 1
    if s1.zone.source == s2.zone.source:
        score += 1.0

    # Cycle 数量相近
    n += 1
    dn = abs(s1.cycle_count - s2.cycle_count)
    score += max(0.0, 1.0 - dn / 3.0)

    # 方向序列一致性
    n += 1
    dir1 = [c.entry.direction.value for c in s1.cycles]
    dir2 = [c.entry.direction.value for c in s2.cycles]
    if dir1 and dir2:
        # 比较截短后的方向序列
        L = min(len(dir1), len(dir2))
        match = sum(1 for i in range(L) if dir1[i] == dir2[i])
        score += match / max(L, 1)
    else:
        score += 0

    return score / n


# ─── 结构族相似 — Label + 镜像判定 ─────────────────────────
def family_similarity(s1: Structure, s2: Structure) -> float:
    if s1.label and s2.label:
        if s1.label == s2.label:
            return 1.0
        if _is_mirror(s1.label, s2.label):
            return 0.6
        return 0.0
    # 无 label，退化为 zone source 一致
    return 0.5 if s1.zone.source == s2.zone.source else 0.2


MIRROR_PAIRS = {
    ("SlowUpFastDown_TopReversal", "FastDownSlowUp_BottomReversal"),
    ("FastUpSlowDown_TopDistribution", "FastDownSlowUp_BottomAccumulation"),
}


def _is_mirror(a: str, b: str) -> bool:
    return (a, b) in MIRROR_PAIRS or (b, a) in MIRROR_PAIRS


# ─── 综合相似度 ────────────────────────────────────────────
def similarity(
        s1: Structure,
        s2: Structure,
        weights: tuple[float, float, float] = (0.4, 0.4, 0.2),
) -> SimilarityScore:
    w_g, w_r, w_f = weights
    g = geometric_similarity(s1, s2)
    r = relational_similarity(s1, s2)
    f = family_similarity(s1, s2)
    total = w_g * g + w_r * r + w_f * f

    matched = {}
    for k in INVARIANT_KEYS:
        v1 = s1.invariants.get(k, 0) or 0
        v2 = s2.invariants.get(k, 0) or 0
        scale = INVARIANT_SCALES[k]
        matched[k] = {
            "s1": v1, "s2": v2,
            "diff_ratio": abs(v1 - v2) / scale if scale > 0 else 0.0,
        }

    return SimilarityScore(
        total=total, geometric=g, relational=r, family=f,
        matched_invariants=matched,
    )
```

```python
# src/retrieval/engine.py
"""
相似结构检索 + 后验统计
"""
from __future__ import annotations
from dataclasses import dataclass
from typing import Sequence
from ..sample.store import Sample, SampleStore
from ..models import Structure
from .similarity import similarity, SimilarityScore


@dataclass
class Neighbor:
    sample: Sample
    score: SimilarityScore


@dataclass
class PosteriorStats:
    sample_size: int
    mean_ret_5d: float
    mean_ret_10d: float
    mean_ret_20d: float
    median_ret_20d: float
    prob_positive_10d: float
    mean_max_dd_20d: float
    mean_max_rise_20d: float


@dataclass
class RetrievalResult:
    query: Structure
    neighbors: list[Neighbor]
    posterior: PosteriorStats


class RetrievalEngine:
    def __init__(self, store: SampleStore):
        self.store = store

    def retrieve(
            self,
            query: Structure,
            top_k: int = 10,
            filter_label: str | None = None,
            min_score: float = 0.5,
    ) -> RetrievalResult:
        candidates = self.store.filter(label_type=filter_label) if filter_label else self.store.load_all()

        # 重建 sample 中的 Structure 骨架供相似度计算
        scored: list[Neighbor] = []
        for sp in candidates:
            s2 = _rebuild_structure_shim(sp)
            if s2 is None:
                continue
            sc = similarity(query, s2)
            if sc.total >= min_score:
                scored.append(Neighbor(sample=sp, score=sc))

        scored.sort(key=lambda n: n.score.total, reverse=True)
        top = scored[:top_k]
        posterior = _aggregate_posterior([n.sample for n in top])
        return RetrievalResult(query=query, neighbors=top, posterior=posterior)


def _rebuild_structure_shim(sp: Sample) -> Structure | None:
    """
    从 Sample.structure (dict) 重建一个最小可用于相似度计算的 Structure
    不重建 Point / Segment 细节，只保留 invariants 与 zone / label
    """
    from ..models import Structure, Zone, ZoneSource
    try:
        zd = sp.structure["zone"]
        zone = Zone(
            price_center=zd["price_center"],
            bandwidth=zd["bandwidth"],
            source=ZoneSource(zd["source"]),
            strength=zd.get("strength", 0.0),
        )
        s = Structure(
            zone=zone,
            invariants=sp.structure.get("invariants", {}),
            label=sp.label_type,
            symbol=sp.symbol,
            typicality=sp.typicality,
        )
        return s
    except Exception:
        return None


def _aggregate_posterior(samples: list[Sample]) -> PosteriorStats:
    n = len(samples)
    if n == 0:
        return PosteriorStats(0, 0, 0, 0, 0, 0, 0, 0)

    def field(name: str) -> list[float]:
        vals = []
        for s in samples:
            if s.forward_outcome and s.forward_outcome.get(name) is not None:
                vals.append(s.forward_outcome[name])
        return vals

    def mean(xs: list[float]) -> float:
        return sum(xs) / len(xs) if xs else 0.0

    def median(xs: list[float]) -> float:
        if not xs:
            return 0.0
        xs = sorted(xs)
        m = len(xs) // 2
        return xs[m] if len(xs) % 2 == 1 else (xs[m - 1] + xs[m]) / 2

    r5, r10, r20 = field("ret_5d"), field("ret_10d"), field("ret_20d")
    dd, rise = field("max_dd_20d"), field("max_rise_20d")
    prob_pos = sum(1 for x in r10 if x > 0) / len(r10) if r10 else 0.0

    return PosteriorStats(
        sample_size=n,
        mean_ret_5d=mean(r5),
        mean_ret_10d=mean(r10),
        mean_ret_20d=mean(r20),
        median_ret_20d=median(r20),
        prob_positive_10d=prob_pos,
        mean_max_dd_20d=mean(dd),
        mean_max_rise_20d=mean(rise),
    )
```

```python
# src/workbench/daily_pipeline.py
"""
每日扫描流水线
1. 读取所有品种数据
2. 编译结构
3. 应用规则扫出候选
4. 检索历史相似
5. 输出日报

from __future__ import annotations
import json
from dataclasses import dataclass, asdict
from datetime import datetime
from pathlib import Path
from typing import Sequence
from ..data.loader import CSVLoader, Bar
from ..compiler.pipeline import compile_structures, CompileConfig
from ..dsl.rule import load_rules, scan, RuleMatch
from ..retrieval.engine import RetrievalEngine
from ..sample.store import SampleStore


@dataclass
class DailyReportItem:
    symbol: str
    structure_label: str
    typicality: float
    t_start: str
    t_end: str
    signature: str
    neighbors: list[dict]
    posterior: dict


def daily_scan(
        data_files: dict[str, str],  # {"CU000": "data/cu0.csv", ...}
        rules_yaml: str = "src/dsl/rules/default.yaml",
        sample_store_path: str = "data/samples/library.jsonl",
        output_dir: str = "output/daily",
        config: CompileConfig | None = None,
        top_k: int = 5,
) -> list[DailyReportItem]:
    cfg = config or CompileConfig()
    rules = load_rules(rules_yaml)
    store = SampleStore(sample_store_path)
    engine = RetrievalEngine(store)

    report: list[DailyReportItem] = []

    for symbol, path in data_files.items():
        loader = CSVLoader(path, symbol=symbol)
        bars = loader.bars
        if not bars:
            continue
        result = compile_structures(bars, cfg, symbol=symbol)
        matches = scan(result.structures, rules)

        for m in matches:
            s = m.structure
            retrieval = engine.retrieve(s, top_k=top_k, filter_label=m.rule.name)
            item = DailyReportItem(
                symbol=symbol,
                structure_label=m.rule.name,
                typicality=m.typicality,
                t_start=s.t_start.isoformat() if s.t_start else "",
                t_end=s.t_end.isoformat() if s.t_end else "",
                signature=s.signature(),
                neighbors=[
                    {
                        "sample_id": n.sample.id,
                        "symbol": n.sample.symbol,
                        "t_start": n.sample.t_start.isoformat(),
                        "score": n.score.total,
                    }
                    for n in retrieval.neighbors
                ],
                posterior=asdict(retrieval.posterior),
            )
            report.append(item)

    # 输出
    out_dir = Path(output_dir)
    out_dir.mkdir(parents=True, exist_ok=True)
    today = datetime.now().strftime("%Y%m%d")
    out_file = out_dir / f"report_{today}.jsonl"
    with open(out_file, "w", encoding="utf-8") as f:
        for item in report:
            f.write(json.dumps(asdict(item), ensure_ascii=False) + "
                                                                   ")
            print(f"日报已输出: {out_file} ({len(report)} 条)")
    return report
```

```python
# scripts/build_sample.py
"""
从历史数据批量生成样本,写入样本库
——
首次运行用来冷启动样本库。
之后也可以定期重跑,扩充样本。
"""
import uuid
from pathlib import Path
import yaml
from src.data.loader import CSVLoader
from src.compiler.pipeline import compile_structures, CompileConfig
from src.dsl.rule import load_rules, scan
from src.sample.store import SampleStore, Sample, sample_from_structure
from src.sample.outcome import compute_forward_outcome


def main():
    with open("config.yaml") as f:
        cfg_all = yaml.safe_load(f)

    compile_cfg = CompileConfig(**cfg_all["compile"])
    rules = load_rules(cfg_all["rules"]["default_yaml"])
    store = SampleStore(cfg_all["samples"]["store_path"])

    data_dir = Path(cfg_all["data"]["csv_dir"])
    for csv_file in sorted(data_dir.glob("*.csv")):
        symbol = csv_file.stem.upper()
        loader = CSVLoader(str(csv_file), symbol=symbol)
        bars = loader.bars
        if len(bars) < 100:
            continue

        result = compile_structures(bars, compile_cfg, symbol=symbol)
        matches = scan(result.structures, rules)

        added = 0
        for m in matches:
            s = m.structure
            if not s.t_end:
                continue
            outcome = compute_forward_outcome(bars, s.t_end)
            sp = sample_from_structure(
                s,
                label_type=m.rule.name,
                sample_id=str(uuid.uuid4())[:8],
                annotation=f"auto from {symbol}",
                forward=outcome,
            )
            store.append(sp)
            added += 1
        print(f"[{symbol}] 新增样本 {added} 条")


if __name__ == "__main__":
    main()
```
